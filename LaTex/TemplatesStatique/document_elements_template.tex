% ================================
% STYLED TABLES
% ================================

% Professional table with Dollarama branding
\newenvironment{DollaramaTable}[2]{%
    \begin{table}[H]
    \centering
    \caption{#1}
    \label{#2}
    \begin{tabular}
}{%
    \end{tabular}
    \end{table}
}

% Long table with Dollarama styling
\newenvironment{DollaramaLongTable}[2]{%
    \begin{longtable}
}{%
    \caption{#1}
    \label{#2}
    \end{longtable}
}

% Financial table style
\newcommand{\financialheader}[1]{%
    \rowcolor{DollaramaGreen!30}
    \textbf{\color{white}#1}
}

% Standard table header
\newcommand{\tableheader}[1]{%
    \rowcolor{DollaramaGreen!20}
    \textbf{#1}
}

% Alternating row colors
\newcommand{\tablerow}[1]{%
    \rowcolor{white}#1
}

\newcommand{\altrow}[1]{%
    \rowcolor{gray!10}#1
}

% ================================
% CITATIONS
% ================================

% Short citation
\newcommand{\citecourt}[2]{%
    \textcolor{DollaramaGreen}{\textbf{[#1]}}%
    \footnote{#2}
}

% Long citation with full reference
\newcommand{\citelong}[4]{%
    \textcolor{DollaramaGreen}{\textbf{[#1]}}%
    \footnote{\textbf{#2} (#3). \textit{#4}}
}

% Internal document citation
\newcommand{\citeinterne}[3]{%
    \textcolor{DollaramaGreen}{\textbf{[DOL-#1]}}%
    \footnote{Dollarama Inc. (#2). \textit{#3}. Document interne.}
}

% Web citation
\newcommand{\citeweb}[4]{%
    \textcolor{DollaramaGreen}{\textbf{[#1]}}%
    \footnote{#2 (#3). \textit{#4}. Consulté le \today}
}

% ================================
% FIGURES FLOTTANTES
% ================================

% Standard figure with Dollarama styling
\newenvironment{DollaramaFigure}[3]{%
    \begin{figure}[#1]
    \centering
    \fcolorbox{DollaramaGreen}{white}{%
    \begin{minipage}{0.95\textwidth}
    \centering
}{%
    \end{minipage}
    }
    \caption{#2}
    \label{#3}
    \end{figure}
}

% Chart figure with border
\newenvironment{DollaramaChart}[3]{%
    \begin{figure}[#1]
    \centering
    \fcolorbox{DollaramaGreen!50}{DollaramaGreen!5}{%
    \begin{minipage}{0.9\textwidth}
    \centering
    \vspace{0.5em}
}{%
    \vspace{0.5em}
    \end{minipage}
    }
    \caption{#2}
    \label{#3}
    \end{figure}
}

% Diagram figure
\newenvironment{DollaramaDiagram}[3]{%
    \begin{figure}[#1]
    \centering
    \begin{tikzpicture}
}{%
    \end{tikzpicture}
    \caption{#2}
    \label{#3}
    \end{figure}
}

% ESPACEMENT VERTICAL

% Small vertical space
\newcommand{\vspacesmall}{\vspace{0.5em}}

% Medium vertical space
\newcommand{\vspacemedium}{\vspace{1em}}

% Large vertical space
\newcommand{\vspacelarge}{\vspace{2em}}

% Extra large vertical space
\newcommand{\vspaceextra}{\vspace{3em}}

% Vertical separator with decorative line
\newcommand{\vseparator}{%
    \vspace{1em}
    \begin{center}
    \textcolor{DollaramaGreen}{\rule{0.3\textwidth}{2pt}}
    \end{center}
    \vspace{1em}
}

% Page break with style
\newcommand{\stylebreak}{%
    \vfill
    \begin{center}
    \textcolor{DollaramaGreen}{\Large $\diamond$}
    \end{center}
    \clearpage
}

% ================================
% AUTHOR AND SIGNATURE BLOCKS
% ================================

% Author block
\newcommand{\authorblock}[4]{%
    \vspace{2em}
    \begin{flushright}
    \begin{tabular}{r}
    \textbf{#1} \\
    \textit{#2} \\
    #3 \\
    #4
    \end{tabular}
    \end{flushright}
}

% Multiple authors
\newcommand{\authorsblock}[1]{%
    \vspace{2em}
    \begin{center}
    \begin{tabular}{c}
    \textbf{\Large Équipe de Rédaction} \\[1em]
    #1
    \end{tabular}
    \end{center}
}

% Signature line
\newcommand{\signatureline}[2]{%
    \vspace{2em}
    \begin{tabular}{p{0.4\textwidth}p{0.2\textwidth}}
    #1 & Date: \rule{3cm}{0.5pt} \\[2em]
    Signature: \rule{5cm}{0.5pt} & \\
    \textbf{#2} & \\
    \end{tabular}
}

% Highlight box
\newcommand{\highlightbox}[1]{%
    \vspace{0.5em}
    \fcolorbox{DollaramaYellow}{DollaramaYellow!20}{%
    \begin{minipage}{0.95\textwidth}
    \vspace{0.3em}
    #1
    \vspace{0.3em}
    \end{minipage}
    }
    \vspace{0.5em}
}

% Warning box
\newcommand{\warningbox}[1]{%
    \vspace{0.5em}
    \fcolorbox{red}{red!10}{%
    \begin{minipage}{0.95\textwidth}
    \vspace{0.3em}
    \textbf{\color{red}⚠ ATTENTION:} #1
    \vspace{0.3em}
    \end{minipage}
    }
    \vspace{0.5em}
}

% Info box
\newcommand{\infobox}[1]{%
    \vspace{0.5em}
    \fcolorbox{DollaramaGreen}{DollaramaGreen!10}{%
    \begin{minipage}{0.95\textwidth}
    \vspace{0.3em}
    \textbf{\color{DollaramaGreen}ℹ INFO:} #1
    \vspace{0.3em}
    \end{minipage}
    }
    \vspace{0.5em}
}

% Quote block
\newcommand{\quotebox}[2]{%
    \vspace{1em}
    \begin{center}
    \fcolorbox{DollaramaGreen!50}{white}{%
    \begin{minipage}{0.8\textwidth}
    \vspace{0.5em}
    \begin{center}
    \textit{\large "#1"}
    \vspace{0.5em}
    
    \textbf{— #2}
    \end{center}
    \vspace{0.5em}
    \end{minipage}
    }
    \end{center}
    \vspace{1em}
}
